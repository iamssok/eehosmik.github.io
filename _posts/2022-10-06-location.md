## location

```html
<form>
    <input type="text" id="psw">
    <input type="button" value="button" onclick="send()">
</form>
```

```javascript
function send() {
    var url = window.location.href;
    var psw = document.getElementById('psw').value;

    if ( psw == '/1234' ) {
        window.location.assign(url + psw);
    } else if ( psw == '' ) {
        alert('비밀번호를 입력하세요.');
    } else {
        alert('비밀번호가 틀렸습니다.');
    }
}
```
